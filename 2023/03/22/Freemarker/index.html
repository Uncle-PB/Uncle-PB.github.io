    简单使用Freemarker生成一个java对象

    包目录结构：
    freemarker
    - JavaBeanAttribute
    - JavaBeanGenerate
    - AutoGenerateCodeUtils
    resources
    - javaBeanGenerate.ftl

    POM-1：free marker

    <dependency>
    <groupId>org.freemarker</groupId>
    <artifactId>freemarker</artifactId>
    <version>2.3.23</version>
</dependency>

    类-1：JavaBean的属性

    @Data
@Builder
public class JavaBeanAttribute {
    // 属性名称
    private String attrName;
    // 类型
    private String attrType;
    // 注释
    private String attrComment;
    // 是否必填
    private String mustRequire;
}

    类-2：生成Java类需要的描述

    @Data
@Builder
public class JavaBeanGenerate {
    // 包名
    private String packageName;
    // 类名
    private String className;
    // 类注释
    private String classComment;
    // 属性
    private List<JavaBeanAttribute> attributes;
    // 需要导入的包
    private List<String> importClassList;
    // 生成的文件需要保存的全路径
    private String path;
}

    类-3：生成代码工具

    AutoGenerateCodeUtils 
    import freemarker.template.Configuration;
import freemarker.template.Template;
import lombok.extern.slf4j.Slf4j;
import qiantao.freeMarker.JavaBeanAttribute;
import qiantao.freeMarker.JavaBeanGenerate;

import java.io.File;
import java.io.FileWriter;
import java.util.*;

@Slf4j
public class AutoGenerateCodeUtils {

    public static void main(String[] args) {
        // 模版配置
        String templatePath = "/Users/admin/Documents/MyCode/fdp/FDP/AA-Test/src/main/resources";
        // 模版名
        String templateName = "javaBeanGenerate.ftl";
        // 输出的包名
        String packageName = "freemarker";
        // 输出位置
        String exportPath = "/Users/admin/Documents/MyCode/fdp/FDP/AA-Test/src/main/java/freemarker";
        // 类名
        String className = "User";
        // 类注释
        String classComment = "用户";

        // 读取定义属性
        JavaBeanAttribute name = JavaBeanAttribute.builder()
                .attrName("name")
                .attrType("String")
                .attrComment("姓名")
                .mustRequire("是否必填：是")
                .build();
        JavaBeanAttribute age = JavaBeanAttribute.builder()
                .attrName("age")
                .attrType("Integer")
                .attrComment("年龄")
                .mustRequire("是否必填：否")
                .build();
        List<JavaBeanAttribute> attributes = Arrays.asList(name, age);

        generateCode(templatePath, templateName, packageName, className, classComment, exportPath, attributes);
    }

    public static void generateCode(String templatePath, String templateName, String packageName, String className,
                                    String classComment, String exportPath, List<JavaBeanAttribute> attributes) {
        // import的类
        Set<String> importList = getImports(attributes);

        // Java bean 生成
        JavaBeanGenerate generate = JavaBeanGenerate.builder()
                .className(className)
                .classComment(classComment)
                .packageName(packageName)
                .attributes(attributes)
                .importClassList(new ArrayList<>(importList))
                .path(exportPath + "/" + className + ".java").build();

        // 模板输出
        fillTemplate(generate.getPath(), templatePath, templateName, generate);
    }

    /**
     * 获取导入的类
     */
    private static Set<String> getImports(List<JavaBeanAttribute> attributes) {
        Set<String> importList = new HashSet<>();
        for (JavaBeanAttribute attribute : attributes) {
            String attrType = attribute.getAttrType();
            if (attrType.equals("Date")) {
                //  date型
                importList.add("java.util.Date");
            } else if (attrType.equals("BigDecimal")) {
                // BigDecimal
                importList.add("java.math.BigDecimal");
            } else if (attrType.startsWith("List")) {
                // List
                importList.add("java.util.List");
            } else if (attrType.startsWith("Map")) {
                // Map
                importList.add("java.util.Map");
            }
        }
        return importList;
    }

    /**
     * 模板输出
     */
    public static void fillTemplate(String outputFilePath, String templatePath, String templateName, Object data) {
        try {
            Configuration cfg = new Configuration();
            cfg.setDirectoryForTemplateLoading(new File(templatePath));

            Template template = cfg.getTemplate(templateName);
            template.process(data, new FileWriter(outputFilePath));

            log.info("生成代码成功");
        } catch (Exception e) {
            log.error("生成代码失败", e);
        }
    }
}

    模版-1：javaBeanGenerate.ftl

    package ${packageName};

<#if importClassList?? && (importClassList?size > 0) >
<#list importClassList as importClass>
import ${importClass};
</#list>
</#if>

import lombok.Getter;
import lombok.Setter;

/**
* ${classComment}
*/
@Getter
@Setter
public class ${className} {

<#if attributes?? && (attributes?size > 0) >
<#list attributes as attr>
    /**
    <#if attr.mustRequire?? >
    *  ${attr.mustRequire}
    </#if>
    <#if attr.attrComment?? >
    *  ${attr.attrComment}
    </#if>
    */
    private ${attr.attrType} ${attr.attrName};

</#list>
</#if>
}
