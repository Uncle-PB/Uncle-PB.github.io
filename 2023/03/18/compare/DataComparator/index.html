    【Compare】增删改数据比较器

    ATM
    发布时间 : 2023-03-18 16:01
    阅读 :
    1. 类：DataComparator
    2. 参考

    类：DataComparator

    import lombok.Data;
import org.apache.commons.collections.CollectionUtils;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.function.Function;
import java.util.stream.Collectors;

@Data
public class DataComparator<T> {
    private List<T> add = new ArrayList<>();
    private List<T> update = new ArrayList<>();
    private List<T> delete = new ArrayList<>();
    private Function<T, String> keyFunction;
    private Compare<T> comparator;

    public void compare(List<T> newList, List<T> originalList) {
        if (CollectionUtils.isEmpty(newList)) {
            delete = originalList;
        }
        Map<String, T> newDataMap = newList.stream().collect(Collectors.toMap(d -> keyFunction.apply(d), Function.identity(), (o1, o2) -> o1));

        for (T oldData : originalList) {
            String key = keyFunction.apply(oldData);
            // 没找到-》删除
            if (!newDataMap.containsKey(key)) {
                delete.add(oldData);
                continue;
            }
            T newData = newDataMap.get(key);
            // 进行比较
            if (!comparator.compare(oldData, newData)) {
                update.add(newData);
            }
            // 从新增中移除
            newDataMap.remove(key);
        }
        // 剩下的去新增
        add.addAll(newDataMap.values());
    }

    interface Compare<T> {
        boolean compare(T t1, T t2);
    }
}
