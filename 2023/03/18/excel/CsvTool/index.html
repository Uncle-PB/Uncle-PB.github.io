    注解：@CsvField

    import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target(ElementType.FIELD)
@Retention(RetentionPolicy.RUNTIME)
public @interface CsvField {
    int index() default 0;
    String name() default "";
    String dateFormat() default "yyyy-MM-dd";
}

    类：CsvTool

    CsvTool 
    
import org.apache.commons.collections.CollectionUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.io.*;
import java.lang.reflect.Field;
import java.text.SimpleDateFormat;
import java.util.*;
import java.util.stream.Collectors;

public class CsvTool {
    private static final Logger logger = LoggerFactory.getLogger(CsvTool.class);
    private BufferedWriter br;

    public CsvTool(OutputStream out) {
        try {
            br = new BufferedWriter(new OutputStreamWriter(out, "GBK"));
        } catch (UnsupportedEncodingException e) {
            logger.error("", e);
        }
    }

    public void writeHeader(Class<?> clazz) {
        if (clazz == null) {
            return;
        }
        try {
            List<String> headers = getHeaders(clazz);
            for (String s : headers) {
                br.write(s);
                br.write(",");
            }
            br.newLine();
        } catch (Exception e) {
            logger.error("", e);
        }
    }

    private List<String> getHeaders(Class<?> clazz) {
        if (clazz == null) {
            return Collections.emptyList();
        }
        Field[] fields = clazz.getDeclaredFields();
        if (fields.length == 0) {
            return Collections.emptyList();
        }
        List<Field> fieldList = Arrays.asList(fields);
        return fieldList.stream().map(f -> f.getAnnotation(CsvField.class))
                .sorted(Comparator.comparing(CsvField::index))
                .map(a -> a.name())
                .collect(Collectors.toList());
    }

    public <T> void writeData(List<T> dataList) {
        if (CollectionUtils.isEmpty(dataList)) {
            return;
        }
        List<List<CsvData>> list = getCsvData(dataList);
        try {
            for (List<CsvData> line : list) {
                for (CsvData s : line) {
                    if (s == null) {
                        br.write("");
                    } else {
                        String writeData = s.getValue();
                        if (writeData == null) {
                            writeData = "";
                        } else if (writeData.contains("\"")) {
                            //处理返回数据包含双引号的特殊数据
                            writeData = writeData.replaceAll("\"", "\"" + "\"");
                        }
                        br.write("\"" + writeData + "\"");
                    }
                    br.write(",");
                }
                br.newLine();
            }
        } catch (Exception e) {
            logger.error("", e);
        }
    }

    public static <T> List<List<CsvData>> getCsvData(List<T> dataList) {
        if (CollectionUtils.isEmpty(dataList)) {
            return Collections.emptyList();
        }
        List<Field> fields = getSortField(dataList.get(0).getClass());
        if (CollectionUtils.isEmpty(fields)) {
            return Collections.emptyList();
        }
        List<List<CsvData>> result = new ArrayList<>();
        for (T t : dataList) {
            try {
                List<CsvData> row = new ArrayList<>();
                for (Field field : fields) {
                    field.setAccessible(true);
                    Object value = field.get(t);
                    CsvField annotation = field.getAnnotation(CsvField.class);
                    row.add(buildCsvData(value, annotation));
                }
                result.add(row);
            } catch (Exception e) {
                logger.error("", e);
            }
        }
        return result;
    }

    private static List<Field> getSortField(Class<?> aClass) {
        Field[] fields = aClass.getDeclaredFields();
        List<Field> list = new ArrayList<>();
        for (Field f : fields) {
            CsvField csvRow = f.getAnnotation(CsvField.class);
            if (csvRow != null) {
                list.add(f);
            }
        }
        Collections.sort(list, (o1, o2) -> {
            CsvField csvRow1 = o1.getAnnotation(CsvField.class);
            CsvField csvRow2 = o2.getAnnotation(CsvField.class);
            return csvRow1.index() - csvRow2.index();
        });
        return list;
    }

    private static CsvData buildCsvData(Object value, CsvField annotation) {
        if (Objects.isNull(value)) {
            return new CsvData("");
        }
        if (value instanceof Date) {
            SimpleDateFormat format = new SimpleDateFormat(annotation.dateFormat());
            String date = format.format((Date) value);
            return new CsvData(date);
        }
        return new CsvData(String.valueOf(value));
    }

    public void close() {
        try {
            br.close();
        } catch (IOException e) {
            logger.error("", e);
        }
    }
}
