    JAVA-1：批处理

    @Autowired
private SqlSessionFactory sqlSessionFactory;
SqlSession session = sqlSessionFactory.openSession(ExecutorType.BATCH, false);
    try {
        CollageUserMapper mapper = session.getMapper(CollageUserMapper.class);
        List<List<CollageUserBO>> partition = Lists.partition(boList, 100);
        for (List<CollageUserBO> list : partition) {
            for (CollageUserBO bo : list) {
                mapper.updateRightStatusById(bo.getId(), bo.getRightsStatus());
            }
            session.commit();
            session.clearCache();
        }
    } catch (Exception e) {
        logger.warn("批量修改预采购单核销队列状态，异常", e);
    } finally {
        session.clearCache();
        session.close();
    }
